  \item \textbf{vt. 刻意低估: to underestimate intentionally}
  \begin{itemize}

  \end{itemize}
  \item \textbf{vt. 表示鄙夷: to express scornfully one's low opinion of}
  \begin{itemize}
    \item \em{\Large{Sore losers minimized the other team's victory. 愤愤不平的失败者对获胜方的胜利表现出鄙夷}}
  \end{itemize}
\end{enumerate}
\end{center}
