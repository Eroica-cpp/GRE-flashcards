  \item \textbf{n. 懒人: a lazy person}
  \begin{itemize}

  \begin{itemize}
    \item \em{\Large{He slouched towards the church as if going to his own funeral. 他缓缓地向教堂走去,仿佛是要去参加自己的葬礼一般。}}
  \end{itemize}
\end{enumerate}
